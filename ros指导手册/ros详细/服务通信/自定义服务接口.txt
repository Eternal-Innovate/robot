1.服务接口 小海龟目标点
$ src/service_interfaces/srv/Patrol.srv
                                                     (默认为cmake 可以不写)
(1)ros2 pkg create service_inter --build-type ament_cmake --dependencies sensor_msgs rosidl_default_generators --license Apache-2.0

(2)创建srv/Patrol.srv

(3)写入
float32 target_x #目标x值
float32 target_y #目标y值
---
int8 SUCCESS = 1 #表示成功
int8 FAIL = 0 #定义常量表示失败
int8 result #处理结果

(4)在CmakeLists
写入
rosidl_generate_interfaces(${PROJECT_NAME}
  "srv/FaceDetector.srv"
  "srv/Patrol.srv"
  DEPENDENCIES sensor_msgs
)

(5)package.xml
写入
<member_of_group>rosidl_interface_packages</member_of_group>
表示属于XX组
(6)
source install/setup.bash
colcon build
(7)ros2 interface show service_inter/srv/Patrol.srv
